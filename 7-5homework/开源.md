void loop() {
  String st="";
  while(Serial.available()>0){
    st[n]=char(Serial.read());
    delay(2);
    n++;
  }
  for(i=0;i<n;i++)
  {
    m.trans(st[i],m);
  }
}

byte income=0;
void loop()
{
 
  if(Serial.available()>0)
  {
    income=Serial.read();
    income=income-'0';
    if(income&0x1)
    {
      digitalWrite(5,HIGH);//这个控制输入1
    }
  else 
  {
    digitalWrite(5,LOW);
  }
  if
    ((income>>1)&0x1)
  {
    digitalWrite(2,HIGH);
  }
  else 
  {
    digitalWrite(2,LOW);
  }
  if((income>>2)&0x1)
  {
    digitalWrite(3,HIGH);
  }
  else 
  {
    digitalWrite(3,LOW);
  }
   if((income>>3)&0x1)

     digitalWrite(4,HIGH);
   }
   else {
     digitalWrite(4,LOW);
   }
  }
    delay(10);
  digitalWrite(0, HIGH);
   delay(10);
}
